22/*
23 * Generic socket address(old but also useful)
24 * struct sockaddr {
25 * sa_family_t sa_family;
26 * char sa_data[14];
27 * }
28 *
29 * IPv4 socket address structure
30 * struct sockaddr_in {
31 * sa_family_t sin_family;
32 * in_port_t sin_port;
33 * struct in_addr sin_addr;
34 * };
35 *
36 * struct in_addr {
37 * in_addr_t s_addr;
38 * };
39 *
40 * IPv6 socket address structure
41 * struct sockaddr_in6 {
42 * sa_family_t sin6_family;
43 * in_port_t sin6_port;
44 * uint32_t sin6_flowinfo;
45 * struct in6_addr sin6_addr;
46 * uint32_t sin6_scoped_id;
47 * };
48 *
49 * struct in6_addr {
50 * unsigned char s6_addr[16];
51 * };
52 *
53 * obviously, the offset of family and port field is same,
54 * so in union, we use these field which no need to distinguish.
55 *
56 * struct addrinfo {
57 * int ai_flags; // Hints argument flags
58 * int ai_family; // specified ip family(1st arg)
59 * int ai_socktype; // specified socket type(2nd arg)
60 * int ai_protocol; // specified protocol(3rd arg)
61 * socklen_t ai_addrlen; // address length
62 * struct sockaddr* ai_addr; // point to the sockaddr structure
63 * char* ai_canonname; // official name
64 * struct addrinfo* ai_next; // point to next item in linked list
65 * };
66 */
