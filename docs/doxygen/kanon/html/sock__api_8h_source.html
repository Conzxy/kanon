79KANON_INLINE void FromIpPort(StringArg ip, uint16_t port,
80 sockaddr_in &addr) KANON_NOEXCEPT
81{
82 addr.sin_family = AF_INET;
83 addr.sin_port = sock::ToNetworkByteOrder16(port);
